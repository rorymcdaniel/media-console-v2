```cpp
#pragma once
#include <cstddef>
#include <cstdint>

class AudioStream
{
public:
    virtual ~AudioStream() = default;

    /// Open the audio source. Returns true on success.
    virtual bool open() = 0;

    /// Close the audio source and release resources.
    virtual void close() = 0;

    /// Read interleaved 16-bit PCM frames into buffer.
    /// Returns number of frames actually read (0 = end of stream, -1 = error).
    virtual long readFrames(int16_t* buffer, size_t frames) = 0;

    /// Total number of frames in the stream (0 if unknown/streaming).
    virtual size_t totalFrames() const = 0;

    /// Current read position in frames.
    virtual size_t positionFrames() const = 0;

    /// Seek to a frame position. Returns true on success.
    virtual bool seek(size_t framePosition) = 0;

    /// Audio format accessors.
    virtual unsigned int sampleRate() const = 0;
    virtual unsigned int channels() const = 0;
    virtual unsigned int bitDepth() const = 0;
};
```

**Step 2: Create AlsaAudioOutput**

Create src/audio/AlsaAudioOutput.h:
```cpp
#pragma once
#include "platform/IAudioOutput.h"

// Forward declare ALSA types to avoid header pollution
typedef struct _snd_pcm snd_pcm_t;

class AlsaAudioOutput : public IAudioOutput
{
public:
    AlsaAudioOutput();
    ~AlsaAudioOutput() override;

    bool open(const QString& deviceName, unsigned int sampleRate, unsigned int channels,
              unsigned int bitDepth) override;
    void close() override;
    void reset() override;
    void pause(bool paused) override;
    long writeFrames(const int16_t* interleaved, size_t frames) override;
    bool isOpen() const override;
    QString deviceName() const override;

private:
    snd_pcm_t* m_handle = nullptr;
    QString m_deviceName;
    bool m_open = false;
};
```

Create src/audio/AlsaAudioOutput.cpp:
- Include `<alsa/asoundlib.h>` in the .cpp file only (not the header)
- open(): Call snd_pcm_open with SND_PCM_STREAM_PLAYBACK, 0 (blocking mode). Set hw_params: access=SND_PCM_ACCESS_RW_INTERLEAVED, format=SND_PCM_FORMAT_S16_LE, channels, rate (use snd_pcm_hw_params_set_rate_near), period_size=1024, buffer_size=8192 (use _near variants). Apply params. On any error, close handle and return false. Log to mediaAudio category.
- close(): If m_handle not null, call snd_pcm_close(m_handle), set m_handle = nullptr, m_open = false.
- reset(): If m_handle not null, call snd_pcm_drop(m_handle) then snd_pcm_prepare(m_handle). This implements "pause cuts audio immediately" per user decision.
- pause(): If paused, try snd_pcm_pause(m_handle, 1). If not supported (-ENOSYS), fall back to snd_pcm_drop + snd_pcm_prepare. If unpausing, snd_pcm_pause(m_handle, 0) or snd_pcm_prepare.
- writeFrames(): Call snd_pcm_writei(m_handle, interleaved, frames). If returns -EPIPE (underrun), log warning, call snd_pcm_prepare, retry snd_pcm_writei. If returns -EIO, log error, return -1 (caller handles recovery). For other negative results, try snd_pcm_recover(m_handle, err, 1), if fails return -1. Return frames written on success.
- isOpen(): return m_open
- deviceName(): return m_deviceName
- Destructor: call close()
