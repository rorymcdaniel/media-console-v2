  <behavior>
    - AudioRingBuffer(352800) creates buffer for 352800 stereo frames (capacity in frames, internally stores frames * 2 int16_t samples)
    - write() on empty buffer with 1024 frames succeeds, returns 1024
    - write() on full buffer returns 0 (no overwrite)
    - read() on empty buffer returns 0
    - read() after write returns the same data that was written
    - availableFrames() returns number of frames ready to read
    - freeFrames() returns number of frames that can be written
    - availableFrames() + freeFrames() == capacity
    - clear() resets read and write positions, availableFrames becomes 0
    - Wraparound: write to near-end, read some, write more (wraps around), read all — data integrity preserved
    - Write more frames than free space returns partial count (fills to capacity)
    - Read more frames than available returns partial count (reads what's available)
    - AudioBufferStats is a POD struct with xrunCount, readLatencyAvgUs, readLatencyMaxUs, errorCount
    - AudioBufferStats reset() zeros all fields

```cpp
#pragma once
#include <cstddef>
#include <cstdint>
#include <vector>

class AudioRingBuffer
{
public:
    explicit AudioRingBuffer(size_t capacityFrames, unsigned int channels = 2);

    /// Write interleaved frames into the buffer. Returns frames actually written.
    size_t write(const int16_t* data, size_t frames);

    /// Read interleaved frames from the buffer. Returns frames actually read.
    size_t read(int16_t* data, size_t frames);

    /// Number of frames available to read.
    size_t availableFrames() const;

    /// Number of frames that can be written.
    size_t freeFrames() const;

    /// Total capacity in frames.
    size_t capacityFrames() const;

    /// Clear all data, reset positions.
    void clear();

private:
    std::vector<int16_t> m_buffer;
    size_t m_readPos = 0;   // In samples (not frames)
    size_t m_writePos = 0;  // In samples (not frames)
    size_t m_count = 0;     // Samples currently in buffer
    size_t m_capacitySamples;
    unsigned int m_channels;
};
```

Implementation: Ring buffer using modular arithmetic on sample positions. write() copies min(frames * channels, free samples) using two memcpy calls if wrapping around end. read() similarly. availableFrames = m_count / m_channels. freeFrames = (m_capacitySamples - m_count) / m_channels.

Create src/audio/LocalPlaybackController.h:
```cpp
#pragma once

#include <QObject>
#include <QThread>

#include <atomic>
#include <chrono>
#include <memory>

class AudioStream;
class IAudioOutput;
class PlaybackState;
class AudioRingBuffer;
struct AudioBufferStats;

class LocalPlaybackController : public QObject
{
    Q_OBJECT

public:
    LocalPlaybackController(IAudioOutput* audioOutput, PlaybackState* playbackState,
                            const QString& deviceName, QObject* parent = nullptr);
    ~LocalPlaybackController() override;

    /// Start playing the given audio stream. Auto-stops any current playback.
    /// Takes shared ownership of the stream.
    void play(std::shared_ptr<AudioStream> stream);

    /// Pause playback — cuts audio immediately (ALSA reset/flush, no drain).
    void pause();

    /// Resume from pause.
    void resume();

    /// Stop playback and close the stream.
    void stop();

    /// Seek to position in milliseconds. Updates UI optimistically.
    void seek(qint64 positionMs);

    /// Returns true if currently playing or paused.
    bool isActive() const;

signals:
    void trackFinished();
    void audioRecoveryFailed();

private:
    void playbackLoop();
    void fillBuffer();
    void handleSeek();
    bool attemptRecovery();
    void updatePosition();
    void stopThread();

    // Helpers
    qint64 framesToMs(size_t frames) const;
    size_t msToFrames(qint64 ms) const;

    IAudioOutput* m_audioOutput;         // Non-owning
    PlaybackState* m_playbackState;      // Non-owning
    QString m_deviceName;

    QThread m_thread;
    std::shared_ptr<AudioStream> m_currentStream;
    std::unique_ptr<AudioRingBuffer> m_ringBuffer;
    std::unique_ptr<AudioBufferStats> m_stats;

    // Atomic control flags — written by main thread, read by playback thread
    std::atomic<bool> m_stopRequested{false};
    std::atomic<bool> m_pauseRequested{false};
    std::atomic<bool> m_seekPending{false};
    std::atomic<qint64> m_seekTargetMs{0};

    // Playback thread state
    size_t m_framesPlayed = 0;
    std::chrono::steady_clock::time_point m_lastPositionUpdate;
    bool m_streamEndReached = false;

    // Named constants from requirements (locked, not configurable)
    static constexpr size_t kBufferFrames = 352800;   // 8 seconds at 44100Hz
    static constexpr size_t kPrefillFrames = 44100;    // 1 second prefill
    static constexpr size_t kPeriodFrames = 1024;      // Read/write chunk size
    static constexpr int kMaxRecoveryRetries = 3;
    static constexpr int kRecoveryBackoffMs = 50;
    static constexpr int kPositionUpdateIntervalMs = 250;
    static constexpr unsigned int kSampleRate = 44100;
    static constexpr unsigned int kChannels = 2;
    static constexpr unsigned int kBitDepth = 16;
};
```

Create src/audio/LocalPlaybackController.cpp:

**Constructor:** Store pointers. Create ring buffer (kBufferFrames, kChannels). Create stats. Do NOT start thread in constructor.

**Destructor:** Call stopThread().

**play(stream):**
1. If thread is running, call stopThread() first (auto-stop current playback).
2. If stream is null, set PlaybackState to Stopped, return.
3. Store stream in m_currentStream.
4. Open the stream. If fails, set Stopped, return.
5. Open audio output: m_audioOutput->open(m_deviceName, kSampleRate, kChannels, kBitDepth). If fails, close stream, set Stopped, return.
6. Calculate duration: `qint64 durationMs = framesToMs(stream->totalFrames())`. Set PlaybackState durationMs and positionMs=0.
7. Reset control flags: m_stopRequested=false, m_pauseRequested=false, m_seekPending=false.
8. Reset m_framesPlayed=0, m_streamEndReached=false, clear ring buffer, reset stats.
9. Set PlaybackState to Playing.
10. Start m_thread with playbackLoop as the entry point. Use `QThread::create([this]{ playbackLoop(); })` pattern — but since m_thread is a member QThread, use the moveToThread pattern: create a worker lambda, connect m_thread.started to it. Actually, simpler approach: override QThread::run is deprecated. Use `std::thread` inside and join in stop? No — use QThread with a worker object pattern. **Simplest correct approach:** Use QThread::create or connect started signal to a lambda that calls playbackLoop. Since we need the thread member for wait(), do:
```cpp
// Connect the thread's started signal to run our loop
connect(&m_thread, &QThread::started, this, [this]() { playbackLoop(); }, Qt::DirectConnection);
m_thread.start();
```
Use Qt::DirectConnection so playbackLoop runs on m_thread, not the main thread. This works because the lambda is invoked when QThread::started fires, which happens on the new thread.

**pause():**
1. m_pauseRequested.store(true, std::memory_order_release).
2. m_audioOutput->reset() — immediate silence per user decision.
3. Set PlaybackState to Paused.

**resume():**
1. m_pauseRequested.store(false, std::memory_order_release).
2. m_audioOutput->pause(false) — unpause ALSA.
3. Set PlaybackState to Playing.

**stop():**
1. Call stopThread().
2. If m_currentStream, close it, reset shared_ptr.
3. m_audioOutput->close().
4. Set PlaybackState to Stopped, positionMs=0.

**seek(positionMs):**
1. Set PlaybackState positionMs immediately (optimistic update per user decision).
2. m_seekTargetMs.store(positionMs, std::memory_order_relaxed).
3. m_seekPending.store(true, std::memory_order_release).

**stopThread():**
1. m_stopRequested.store(true, std::memory_order_release).
2. m_thread.quit().
3. m_thread.wait(5000). If not finished, terminate and warn.
4. Disconnect the started signal to prevent re-entry on next play().

**playbackLoop():**
```
Prefill ring buffer (fill to kPrefillFrames).

while (!m_stopRequested):
    if m_seekPending:
        handleSeek()
        continue

    if m_pauseRequested:
        QThread::msleep(10)
        continue

    fillBuffer()

    if m_ringBuffer->availableFrames() == 0 and m_streamEndReached:
        // End of track
        QMetaObject::invokeMethod(this, [this](){
            m_playbackState->setPlaybackMode(PlaybackMode::Stopped);
            emit trackFinished();
        }, Qt::QueuedConnection);
        break

    chunk = min(m_ringBuffer->availableFrames(), kPeriodFrames)
    if chunk == 0: continue  // Buffer empty but stream not ended, wait for fill

    Read from ring buffer into local array.
    auto start = steady_clock::now()
    written = m_audioOutput->writeFrames(data, chunk)
    auto elapsed = duration_cast<microseconds>(steady_clock::now() - start)

    if written < 0:
        if !attemptRecovery():
            QMetaObject::invokeMethod(this, [this](){
                m_playbackState->setPlaybackMode(PlaybackMode::Stopped);
                emit audioRecoveryFailed();
            }, Qt::QueuedConnection);
            break
        continue  // Retry after recovery

    if written > 0:
        m_framesPlayed += written
        // Check for underrun if written < chunk
        m_stats->recordReadLatency(elapsed.count())

    updatePosition()
```

**fillBuffer():**
```
while m_ringBuffer->freeFrames() >= kPeriodFrames and !m_streamEndReached:
    Read kPeriodFrames from m_currentStream->readFrames into temp buffer
    if readResult == 0:
        m_streamEndReached = true
        break
    if readResult < 0:
        m_stats->recordError()
        break
    m_ringBuffer->write(tempBuffer, readResult)
```

**handleSeek():**
1. m_seekPending.store(false, std::memory_order_release).
2. size_t targetFrame = msToFrames(m_seekTargetMs.load()).
3. m_currentStream->seek(targetFrame).
4. m_ringBuffer->clear().
5. m_framesPlayed = targetFrame.
6. m_streamEndReached = false.
7. m_audioOutput->reset() — flush old audio.
8. Prefill ring buffer to kPrefillFrames before resuming.
9. m_audioOutput->pause(false) if was paused — actually just snd_pcm_prepare via the output.

**attemptRecovery():**
```
for retry = 0 to kMaxRecoveryRetries:
    m_audioOutput->close()
    QThread::msleep(kRecoveryBackoffMs)
    if m_audioOutput->open(m_deviceName, kSampleRate, kChannels, kBitDepth):
        m_stats->recordXrun()
        Log recovery success
        return true
    m_stats->recordError()
Log all retries exhausted
return false
```

**updatePosition():**
```
auto now = steady_clock::now()
if duration_cast<milliseconds>(now - m_lastPositionUpdate).count() >= kPositionUpdateIntervalMs:
    qint64 posMs = framesToMs(m_framesPlayed)
    QMetaObject::invokeMethod(this, [this, posMs](){
        m_playbackState->setPositionMs(posMs);
    }, Qt::QueuedConnection);
    m_lastPositionUpdate = now
```

**Helper methods:**
- framesToMs: `return static_cast<qint64>(frames * 1000 / kSampleRate);`
- msToFrames: `return static_cast<size_t>(ms * kSampleRate / 1000);`
