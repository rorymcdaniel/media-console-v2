**Step 2: Create LibcdioCdDrive**

Create src/cd/LibcdioCdDrive.h:
```cpp
#pragma once
#include "platform/ICdDrive.h"

// Forward declarations to avoid libcdio header pollution
typedef struct _CdIo CdIo_t;

class LibcdioCdDrive : public ICdDrive
{
public:
    LibcdioCdDrive();
    ~LibcdioCdDrive() override;

    bool openDevice(const QString& devicePath) override;
    QVector<TocEntry> readToc() override;
    QString getDiscId() override;
    bool eject() override;
    bool stopSpindle() override;
    bool isDiscPresent() const override;
    bool isAudioDisc() const override;
    int trackCount() const override;

private:
    CdIo_t* m_cdio = nullptr;
    QString m_devicePath;
    int m_trackCount = 0;
};
```

Create src/cd/LibcdioCdDrive.cpp:
- Include `<cdio/cdio.h>`, `<cdio/mmc.h>`, `<discid/discid.h>`, and Linux headers `<linux/cdrom.h>`, `<sys/ioctl.h>`, `<fcntl.h>` (under `#ifdef __linux__`)
- openDevice(): Call `cdio_open(path, DRIVER_DEVICE)`. Store handle. Store devicePath for ioctl calls. Log to mediaCd.
- readToc(): Use `cdio_get_first_track_num(m_cdio)` and `cdio_get_num_tracks(m_cdio)`. For each track, get LBA via `cdio_get_track_lba(m_cdio, trackNum)` for start sector. End sector = next track's start - 1 (use leadout for last track via `CDIO_CDROM_LEADOUT_TRACK`). Duration = (endSector - startSector + 1) / 75 (75 sectors per second for CD audio). Store m_trackCount.
- getDiscId(): Create `DiscId* disc = discid_new()`. Use `discid_read(disc, m_devicePath.toLocal8Bit().data())`. If success, get ID via `discid_get_id(disc)`. Free with `discid_free(disc)`. Return empty QString on failure.
- isDiscPresent(): Open device fd with O_RDONLY | O_NONBLOCK. ioctl(fd, CDROM_DRIVE_STATUS, CDSL_CURRENT). Close fd. Return status == CDS_DISC_OK. On non-Linux, return false.
- isAudioDisc(): Same fd pattern. ioctl(fd, CDROM_DISC_STATUS). Return discType == CDS_AUDIO || discType == CDS_MIXED.
- eject(): `cdio_eject_media(&m_cdio)`. Note: this NULLs m_cdio.
- stopSpindle(): Use `cdio_set_speed(m_cdio, 0)` to spin down. If that fails, try ioctl START_STOP with stop flag.
- trackCount(): return m_trackCount.
- Destructor: if m_cdio, call `cdio_destroy(m_cdio)`.

Create src/cd/CdAudioStream.h:
```cpp
#pragma once
#include "audio/AudioStream.h"
#include <QString>

// Forward declarations
typedef struct cdrom_drive_s cdrom_drive_t;
typedef struct cdrom_paranoia_s cdrom_paranoia_t;

class CdAudioStream : public AudioStream
{
public:
    CdAudioStream(const QString& devicePath, int startSector, int endSector);
    ~CdAudioStream() override;

    bool open() override;
    void close() override;
    long readFrames(int16_t* buffer, size_t frames) override;
    size_t totalFrames() const override;
    size_t positionFrames() const override;
    bool seek(size_t framePosition) override;
    unsigned int sampleRate() const override { return 44100; }
    unsigned int channels() const override { return 2; }
    unsigned int bitDepth() const override { return 16; }

private:
    static constexpr int kFramesPerSector = 588;
    static constexpr int kBytesPerFrame = 4;  // 16-bit stereo = 4 bytes

    QString m_devicePath;
    int m_startSector;
    int m_endSector;
    int m_currentSector;

    cdrom_drive_t* m_cdda = nullptr;
    cdrom_paranoia_t* m_paranoia = nullptr;

    // Partial sector buffer for readFrames that don't align to sector boundaries
    int16_t m_sectorBuffer[588 * 2] = {};  // One sector of stereo samples
    int m_sectorBufferOffset = 0;           // Frames remaining in buffer
    int m_sectorBufferFrames = 0;           // Total frames available in buffer
};
```

Create src/cd/CdAudioStream.cpp:
- Include `<cdio/paranoia/cdda.h>` and `<cdio/paranoia/paranoia.h>`
- Constructor: store devicePath, startSector, endSector. Set m_currentSector = startSector.
- open():
  - `m_cdda = cdio_cddap_identify(m_devicePath.toLocal8Bit().data(), CDDA_MESSAGE_FORGETIT, nullptr)`
  - If null, log error and return false
  - `int rc = cdio_cddap_open(m_cdda)` -- if nonzero, log error, clean up, return false
  - `m_paranoia = cdio_paranoia_init(m_cdda)`
  - `cdio_paranoia_modeset(m_paranoia, PARANOIA_MODE_FULL)` for maximum error correction
  - `cdio_paranoia_seek(m_paranoia, m_startSector, SEEK_SET)`
  - m_currentSector = m_startSector
  - m_sectorBufferOffset = 0; m_sectorBufferFrames = 0
  - return true
- readFrames(buffer, frames):
  - Track framesRead = 0
  - First: drain any remaining frames from m_sectorBuffer
  - Then: while framesRead < frames && m_currentSector < m_endSector:
    - `int16_t* sector = cdio_paranoia_read_limited(m_paranoia, nullptr, 10)` (10 retries)
    - If sector is null, return -1 (read error)
    - Copy sector data into output buffer. Each sector = kFramesPerSector stereo frames.
    - If we need fewer frames than a full sector, buffer the remainder in m_sectorBuffer
    - m_currentSector++ after full sector consumed
    - Update framesRead
  - return framesRead (0 means end of stream)
- seek(framePosition):
  - Convert frame position to sector: sector = m_startSector + (framePosition / kFramesPerSector)
  - If sector >= m_endSector, return false
  - `cdio_paranoia_seek(m_paranoia, sector, SEEK_SET)`
  - m_currentSector = sector
  - Handle sub-sector offset: m_sectorBufferOffset for frames within the sector
  - return true
- totalFrames(): return (m_endSector - m_startSector) * kFramesPerSector
- positionFrames(): return (m_currentSector - m_startSector) * kFramesPerSector - m_sectorBufferFrames + m_sectorBufferOffset (accounting for buffered partial sectors)
- close(): cdio_paranoia_free(m_paranoia), cdio_cddap_close(m_cdda), set both to nullptr
- Destructor: call close()
