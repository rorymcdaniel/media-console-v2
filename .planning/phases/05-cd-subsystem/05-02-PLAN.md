```cpp
#pragma once
#include <QObject>
#include <QString>
#include <QVector>
#include <optional>

struct CdTrackInfo
{
    int trackNumber = 0;
    QString title;
    QString artist;
    int durationSeconds = 0;
};

struct CdMetadata
{
    QString discId;
    QString artist;
    QString album;
    QString genre;
    int year = 0;
    QString source;  // "musicbrainz", "gnudb", "discogs"
    QString musicbrainzReleaseId;
    QVector<CdTrackInfo> tracks;
};

struct CdAlbumArtInfo
{
    QString frontPath;
    QString backPath;
    QString source;  // "coverartarchive", "discogs"
};

class CdMetadataCache : public QObject
{
    Q_OBJECT
public:
    explicit CdMetadataCache(QObject* parent = nullptr);
    ~CdMetadataCache() override;

    bool open(const QString& dbPath);
    void close();

    std::optional<CdMetadata> lookup(const QString& discId);
    bool store(const CdMetadata& metadata);

    std::optional<CdAlbumArtInfo> getAlbumArt(const QString& discId);
    bool storeAlbumArt(const QString& discId, const QString& frontPath,
                        const QString& backPath, const QString& source);

private:
    bool createTables();
    QString m_connectionName;
};
```

Create src/cd/CdMetadataCache.cpp:
- Use `QSqlDatabase::addDatabase("QSQLITE", m_connectionName)` with a unique connection name (e.g., "cd_metadata_cache") to avoid default connection conflicts.
- open(): Set database path, open connection, call createTables().
- createTables(): Execute CREATE TABLE IF NOT EXISTS for disc_metadata, track_metadata, album_art tables per schema in RESEARCH.md. Use QSqlQuery with the named connection.
- lookup(): SELECT from disc_metadata WHERE disc_id = ?. If found, SELECT tracks from track_metadata. Build CdMetadata and return.
- store(): Use INSERT OR REPLACE into disc_metadata. DELETE existing track_metadata for this disc_id, then INSERT new tracks. Wrap in transaction for atomicity.
- storeAlbumArt(): INSERT OR REPLACE into album_art.
- getAlbumArt(): SELECT from album_art WHERE disc_id = ?.
- close(): QSqlDatabase::removeDatabase(m_connectionName).
- Destructor: call close().
- All queries use prepared statements (QSqlQuery::prepare + bindValue) to prevent SQL injection.
- Log errors to mediaCd category.

**Step 1: Create CdMetadataProvider**

Create src/cd/CdMetadataProvider.h:
```cpp
#pragma once
#include <QObject>
#include <QVector>
#include "cd/CdMetadataCache.h"  // for CdMetadata, CdTrackInfo

class QNetworkAccessManager;
class QNetworkReply;

class CdMetadataProvider : public QObject
{
    Q_OBJECT
public:
    explicit CdMetadataProvider(QNetworkAccessManager* nam, QObject* parent = nullptr);

    void lookup(const QString& discId, const QVector<TocEntry>& toc);
    void cancel();

signals:
    void metadataReady(const CdMetadata& metadata);
    void lookupFailed(const QString& discId);

private:
    void lookupMusicBrainz();
    void lookupGnuDb();
    void lookupDiscogs();

    bool parseMusicBrainzResponse(const QByteArray& data);
    bool parseGnuDbResponse(const QByteArray& data);
    bool parseDiscogsResponse(const QByteArray& data);

    // GnuDB validation: reject if response track count != TOC track count
    bool validateGnuDbTrackCount(int responseTrackCount);

    // Build GnuDB query URL from TOC data
    QString buildGnuDbQueryUrl();
    QString buildGnuDbReadUrl(const QString& category, const QString& discId);

    QNetworkAccessManager* m_nam;  // Non-owning
    QString m_discId;
    QVector<TocEntry> m_toc;
    CdMetadata m_metadata;
    QNetworkReply* m_activeReply = nullptr;

    static constexpr int kRequestTimeoutMs = 10000;  // 10 second timeout per request
};
```

Create src/cd/CdMetadataProvider.cpp:
- Constructor: store QNetworkAccessManager pointer.
- lookup(): Store discId and toc. Initialize m_metadata with discId. Start with lookupMusicBrainz().
- cancel(): If m_activeReply, abort and deleteLater.
- lookupMusicBrainz():
  - URL: `https://musicbrainz.org/ws/2/discid/{discId}?fmt=json&inc=recordings+artists`
  - Request: Set "User-Agent" header to "MediaConsole/2.0 (https://github.com/rory)" (MusicBrainz TOS requires User-Agent)
  - On success (200): Parse JSON. Extract first release: title -> album, artist-credit -> artist, media[].track-list -> track titles. Extract release ID for CoverArtArchive. Emit metadataReady.
  - On failure (non-200 or parse error): Call lookupGnuDb().
- lookupGnuDb():
  - Build query URL using CDDB protocol over HTTP: `http://gnudb.gnudb.org/~cddb/cddb.cgi?cmd=cddb+query+{freedb_disc_id}+{ntrks}+{offsets}+{nsecs}&hello=user+hostname+MediaConsole+2.0&proto=6`
  - Note: GnuDB uses a different disc ID format than MusicBrainz. Compute freedb disc ID from TOC: sum of digit sums of start seconds for each track, mod 255, shifted left 24 bits, OR'd with total seconds shifted left 8 bits, OR'd with track count.
  - On query success: Parse response for category and disc ID. Make second request to read full entry: `cmd=cddb+read+{category}+{discid}`
  - Parse DTITLE (artist / album split by " / "), TTITLE0..TTITLEn for track titles.
  - Validate: response track count must match m_toc.size(). If mismatch, reject and fall through to Discogs (per CD-09).
  - Try UTF-8 decoding first; if invalid, fall back to Latin-1 (per research pitfall).
  - On success: set source="gnudb", emit metadataReady.
  - On failure: call lookupDiscogs().
- lookupDiscogs():
  - Search by disc title/artist if available from prior attempt, or by track count + durations.
  - URL: `https://api.discogs.com/database/search?type=release&barcode=&format=CD&track={trackCount}&per_page=5`
  - Note: Discogs requires authentication. For now, include User-Agent and accept first reasonable match.
  - On success: Parse JSON, extract title, artist, year. Set source="discogs".
  - On failure: Emit lookupFailed. The CdController will apply "Audio CD" fallback per user decision.
- All requests use QNetworkRequest with timeout via QTimer::singleShot connected to reply->abort().

```cpp
#pragma once
#include <QObject>

class QNetworkAccessManager;

class CdAlbumArtProvider : public QObject
{
    Q_OBJECT
public:
    explicit CdAlbumArtProvider(QNetworkAccessManager* nam, const QString& cacheDir,
                                 QObject* parent = nullptr);

    void download(const QString& musicbrainzReleaseId, const QString& discId);
    void cancel();

signals:
    void albumArtReady(const QString& discId, const QString& frontPath, const QString& backPath);
    void albumArtFailed(const QString& discId);

private:
    void downloadFromCoverArtArchive(const QString& releaseId, const QString& discId);
    void downloadFromDiscogs(const QString& discId);
    QString saveToDisk(const QString& discId, const QString& type, const QByteArray& data);

    QNetworkAccessManager* m_nam;  // Non-owning
    QString m_cacheDir;
};
```

Create src/cd/CdAlbumArtProvider.cpp:
- Constructor: store NAM pointer and cache directory. Create cache dir if not exists (QDir::mkpath).
- download(): If musicbrainzReleaseId is non-empty, try CoverArtArchive first. Otherwise, try Discogs directly.
- downloadFromCoverArtArchive():
  - Front: GET `https://coverartarchive.org/release/{releaseId}/front`
  - Back: GET `https://coverartarchive.org/release/{releaseId}/back`
  - CoverArtArchive returns 307 redirect to Internet Archive -- QNetworkAccessManager follows redirects automatically.
  - On success: Save to disk via saveToDisk(). Emit albumArtReady with paths.
  - On failure (404 = no art): Try Discogs as fallback per user decision.
- downloadFromDiscogs(): Placeholder implementation -- requires Discogs API key. For now, emit albumArtFailed if CoverArtArchive failed.
- saveToDisk(): Save to `{cacheDir}/{discId}_{type}.jpg`. Return path. Detect image format from data (JPEG magic bytes vs PNG).
