  <behavior>
    - CdController::start() begins disc polling via QTimer at CdConfig::pollIntervalMs
    - Polling detects disc insertion (transition from no-disc to disc-present)
    - On disc insertion: emit discDetected(), read TOC, emit tocReady(toc)
    - After TOC: check cache. If hit, emit metadataReady immediately. If miss, start async lookup.
    - After metadata lookup success: store in cache, emit metadataReady
    - After metadata lookup failure: set "Audio CD" fallback, emit metadataReady (no error signal)
    - After metadata: start album art download if MusicBrainz release ID available
    - On disc removal: emit discRemoved(), cancel pending lookups, stop active playback
    - Idle timer: starts after playback stops, reset on any CD activity, fires stopSpindle after timeout
    - playTrack(trackNumber): creates CdAudioStream, handles spin-up if needed, calls LocalPlaybackController::play()
    - Non-audio disc: emit nonAudioDiscDetected() for toast, no further action
    - No auto-play: discDetected and tocReady are informational only

Create src/cd/CdController.h:
```cpp
#pragma once
#include <QObject>
#include <QTimer>
#include <QVector>
#include <memory>

#include "cd/CdMetadataCache.h"  // for CdMetadata, CdTrackInfo, CdAlbumArtInfo

struct TocEntry;
struct CdConfig;
class ICdDrive;
class CdMetadataProvider;
class CdAlbumArtProvider;
class LocalPlaybackController;
class PlaybackState;
class QNetworkAccessManager;

class CdController : public QObject
{
    Q_OBJECT

public:
    CdController(ICdDrive* drive, LocalPlaybackController* playbackController,
                  PlaybackState* playbackState, const CdConfig& config, QObject* parent = nullptr);
    ~CdController() override;

    void start();
    void stop();

    // User-initiated playback (CD-04: ALWAYS user-initiated, never auto-play)
    void playTrack(int trackNumber);
    void eject();

    // Accessors for UI binding
    QVector<TocEntry> currentToc() const;
    CdMetadata currentMetadata() const;
    bool isDiscPresent() const;

signals:
    // Progressive display signals (CD-06)
    void discDetected();                    // "Reading disc..." state
    void tocReady(const QVector<TocEntry>& toc);   // Track listing with durations
    void metadataReady(const CdMetadata& metadata); // Titles/artist/album filled in
    void albumArtReady(const QString& frontPath, const QString& backPath); // Art cached

    void discRemoved();                     // Clear everything
    void nonAudioDiscDetected();            // Toast: "Not an audio disc"

    void spinUpStarted();                   // Play button -> loading state
    void spinUpComplete();                  // Playback starts

private slots:
    void onPollTimer();
    void onMetadataReady(const CdMetadata& metadata);
    void onMetadataFailed(const QString& discId);
    void onAlbumArtReady(const QString& discId, const QString& frontPath, const QString& backPath);
    void onAlbumArtFailed(const QString& discId);
    void onTrackFinished();
    void onIdleTimeout();

private:
    void handleDiscInserted();
    void handleDiscRemoved();
    void startIdleTimer();
    void resetIdleTimer();
    void stopIdleTimer();
    void startSpinUpCheck(int trackNumber);

    ICdDrive* m_drive;  // Non-owning
    LocalPlaybackController* m_playbackController;  // Non-owning
    PlaybackState* m_playbackState;  // Non-owning
    CdConfig m_config;

    std::unique_ptr<QNetworkAccessManager> m_nam;
    std::unique_ptr<CdMetadataCache> m_cache;
    std::unique_ptr<CdMetadataProvider> m_metadataProvider;
    std::unique_ptr<CdAlbumArtProvider> m_albumArtProvider;

    QTimer m_pollTimer;
    QTimer m_idleTimer;
    QTimer m_spinUpTimer;

    bool m_discPresent = false;
    bool m_isSpinningUp = false;
    int m_pendingTrack = -1;

    QVector<TocEntry> m_currentToc;
    CdMetadata m_currentMetadata;
    QString m_currentDiscId;
};
```

Create src/cd/CdController.cpp:
- Constructor:
  - Store non-owning pointers (drive, playbackController, playbackState)
  - Store config
  - Create m_nam = make_unique<QNetworkAccessManager>(this)
  - Create m_cache = make_unique<CdMetadataCache>(this). Open at `{QStandardPaths::writableLocation(QStandardPaths::CacheLocation)}/cd_metadata.db`
  - Create m_metadataProvider = make_unique<CdMetadataProvider>(m_nam.get(), this)
  - Create m_albumArtProvider = make_unique<CdAlbumArtProvider>(m_nam.get(), `{cache}/albumart`, this)
  - Connect signals: m_metadataProvider->metadataReady -> onMetadataReady, lookupFailed -> onMetadataFailed
  - Connect: m_albumArtProvider->albumArtReady -> onAlbumArtReady, albumArtFailed -> onAlbumArtFailed
  - Connect: m_playbackController->trackFinished -> onTrackFinished
  - Connect: m_pollTimer.timeout -> onPollTimer
  - Connect: m_idleTimer.timeout -> onIdleTimeout
  - Configure idle timer: m_idleTimer.setSingleShot(true), setInterval(config.idleTimeoutSeconds * 1000)

- start(): Open device via m_drive->openDevice(m_config.devicePath). Start poll timer at m_config.pollIntervalMs.

- onPollTimer():
  - Check m_drive->isDiscPresent()
  - If disc newly inserted (was false, now true): handleDiscInserted()
  - If disc removed (was true, now false): handleDiscRemoved()

- handleDiscInserted():
  - m_discPresent = true
  - emit discDetected()  // UI shows "Reading disc..."
  - Check m_drive->isAudioDisc(). If not and m_config.audioOnly: emit nonAudioDiscDetected(), return
  - Read TOC: m_currentToc = m_drive->readToc()
  - emit tocReady(m_currentToc)  // UI shows track listing with durations
  - Get disc ID: m_currentDiscId = m_drive->getDiscId()
  - Check cache: auto cached = m_cache->lookup(m_currentDiscId)
  - If cached.has_value():
    - m_currentMetadata = cached.value()
    - emit metadataReady(m_currentMetadata)  // Instant! Per user decision for reinsertion
    - Check art cache: auto art = m_cache->getAlbumArt(m_currentDiscId)
    - If art: emit albumArtReady(art->frontPath, art->backPath)
    - Else: trigger art download
  - Else: m_metadataProvider->lookup(m_currentDiscId, m_currentToc)
  - Start idle timer

- handleDiscRemoved():
  - m_discPresent = false
  - m_metadataProvider->cancel()
  - m_albumArtProvider->cancel()
  - If m_playbackController->isActive(): m_playbackController->stop()
  - Clear state: m_currentToc.clear(), m_currentMetadata = {}, m_currentDiscId.clear()
  - emit discRemoved()  // UI clears track listing
  - Stop idle timer

- onMetadataReady(metadata):
  - m_currentMetadata = metadata
  - m_cache->store(metadata)
  - emit metadataReady(metadata)  // UI batch-updates all titles
  - If metadata.musicbrainzReleaseId not empty: m_albumArtProvider->download(metadata.musicbrainzReleaseId, metadata.discId)

- onMetadataFailed(discId):
  - Per user decision: silently show "Audio CD"
  - m_currentMetadata.discId = discId
  - m_currentMetadata.artist = "Audio CD"
  - m_currentMetadata.album = "Audio CD"
  - m_currentMetadata.source = "fallback"
  - emit metadataReady(m_currentMetadata)
  - No error indicator, no toast per user decision

- onAlbumArtReady(discId, frontPath, backPath):
  - m_cache->storeAlbumArt(discId, frontPath, backPath, "coverartarchive")
  - emit albumArtReady(frontPath, backPath)

- onAlbumArtFailed(discId):
  - Silent failure. No error indicator. Art area just stays empty.

- playTrack(trackNumber):
  - If trackNumber < 1 or > m_currentToc.size(), return
  - resetIdleTimer()
  - const auto& track = m_currentToc[trackNumber - 1]
  - auto stream = std::make_shared<CdAudioStream>(m_config.devicePath, track.startSector, track.endSector)
  - Update PlaybackState: setActiveSource(MediaSource::Cd), setTrackNumber(trackNumber), setTrackCount(m_currentToc.size())
  - Set track metadata from m_currentMetadata if available
  - m_playbackController->play(stream)
  - Note: If drive needs spin-up, CdAudioStream::open() may block on the playback thread until drive is ready. The LocalPlaybackController handles this on its background thread, so no main thread blocking. For spin-up UI feedback, emit spinUpStarted before play() and connect to playback start to emit spinUpComplete. (Simplified approach: playback thread blocks during spin-up; UI button already shows loading because PlaybackState transitions to Playing asynchronously.)

- onTrackFinished():
  - Start idle timer

- onIdleTimeout():
  - m_drive->stopSpindle()

- eject():
  - If m_playbackController->isActive(): m_playbackController->stop()
  - m_drive->eject()
  - handleDiscRemoved()  // Will emit discRemoved, clear state
