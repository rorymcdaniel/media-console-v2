Create src/library/LibraryScanner.h:
```cpp
#pragma once

#include <QObject>
#include <QFutureWatcher>
#include <QMap>
#include <QVector>

#include <atomic>

struct LibraryTrack;

/// Recursive FLAC file scanner with TagLib metadata extraction.
/// Runs asynchronously via QtConcurrent. Emits batches of scanned tracks
/// for main thread DB insertion. Supports incremental scanning via mtime comparison.
class LibraryScanner : public QObject
{
    Q_OBJECT
public:
    explicit LibraryScanner(QObject* parent = nullptr);
    ~LibraryScanner() override;

    /// Start async scan. existingMtimes maps filePath -> mtime for skip logic.
    void startScan(const QString& rootPath, const QMap<QString, qint64>& existingMtimes);

    /// Request cancellation. Scan stops at next file boundary.
    void cancel();

    /// Returns true if scan is currently running.
    bool isScanning() const;

signals:
    void batchReady(const QVector<LibraryTrack>& tracks);
    void scanProgress(int processed, int total);
    void scanComplete(int totalProcessed, int totalSkipped);
    void scanError(const QString& filePath, const QString& error);

private:
    void doScan(const QString& rootPath, const QMap<QString, qint64>& existingMtimes);
    bool extractMetadata(const QString& filePath, LibraryTrack& track);

    QFutureWatcher<void> m_watcher;
    std::atomic<bool> m_cancelled{false};

    static constexpr int kBatchSize = 50;
};
```

Create src/library/LibraryScanner.cpp:
- Include `<QDirIterator>`, `<QFileInfo>`, `<QtConcurrent>`, and TagLib headers:
  ```cpp
  #include <taglib/fileref.h>
  #include <taglib/flacfile.h>
  #include <taglib/tag.h>
  #include <taglib/tpropertymap.h>
  ```
- startScan():
  ```cpp
  m_cancelled.store(false);
  auto future = QtConcurrent::run([this, rootPath, existingMtimes]() {
      doScan(rootPath, existingMtimes);
  });
  m_watcher.setFuture(future);
  ```
- doScan():
  - First pass: count FLAC files with QDirIterator for progress total
  - Second pass: iterate with QDirIterator(rootPath, {"*.flac"}, QDir::Files, QDirIterator::Subdirectories)
  - For each file:
    - Check m_cancelled.load() -- return if true
    - Get mtime via QFileInfo::lastModified().toSecsSinceEpoch()
    - If existingMtimes.value(filePath, -1) == mtime, skip (increment skipped counter)
    - Else: call extractMetadata(filePath, track); if success, add to batch
    - If batch.size() >= kBatchSize, emit batchReady(batch), clear batch
    - emit scanProgress(processed, total)
  - After loop: emit remaining batch if non-empty
  - emit scanComplete(totalProcessed, totalSkipped)
- extractMetadata():
  - `TagLib::FileRef fileRef(filePath.toUtf8().constData());`
  - if fileRef.isNull(): emit scanError, return false
  - if !fileRef.tag(): emit scanError, return false
  - Extract from tag(): title, artist, album, genre, year, trackNumber
  - Extract from properties(): ALBUMARTIST (fallback to artist), DISCNUMBER (parse "1/2" format, take first part)
  - Extract from audioProperties(): durationSeconds (lengthInSeconds), sampleRate, bitDepth (bitsPerSample)
  - Set track.mtime from QFileInfo
  - return true
- cancel(): m_cancelled.store(true)
- isScanning(): return m_watcher.isRunning()

Create src/library/LibraryAlbumArtProvider.h:
```cpp
#pragma once

#include <QObject>
#include <QString>

struct LibraryAlbumArt
{
    QString frontPath;
    QString backPath;
};

/// Extracts album art from FLAC picture blocks and caches to disk.
/// Falls back to cover.jpg/folder.jpg/front.jpg in album directory.
/// Cache filenames use SHA-1(albumArtist + album) per FLAC-05.
class LibraryAlbumArtProvider : public QObject
{
    Q_OBJECT
public:
    explicit LibraryAlbumArtProvider(const QString& cacheDir, QObject* parent = nullptr);

    /// Extract art for an album. Can be called from background thread.
    /// Returns paths to cached front/back art (empty if not found).
    LibraryAlbumArt extractArt(const QString& flacPath, const QString& albumArtist, const QString& album);

    /// Check if art is already cached for this album.
    bool hasCachedArt(const QString& albumArtist, const QString& album) const;

    /// Get cached art paths (empty if not cached).
    LibraryAlbumArt getCachedArt(const QString& albumArtist, const QString& album) const;

private:
    QString computeHash(const QString& albumArtist, const QString& album) const;
    QString extractFromFlac(const QString& flacPath, const QString& hash, const QString& type);
    QString findFileArt(const QString& albumDir, const QString& hash, const QString& type);
    QString saveToDisk(const QString& hash, const QString& type, const QByteArray& data);
    QString detectExtension(const QByteArray& data) const;

    QString m_cacheDir;

    // Fallback filenames to scan in album directory
    static const QStringList kFrontFallbacks;
    static const QStringList kBackFallbacks;
};
```

Create src/library/LibraryAlbumArtProvider.cpp:
- Include TagLib headers: `<taglib/flacfile.h>`, `<taglib/flacpicture.h>`
- Include Qt: `<QCryptographicHash>`, `<QDir>`, `<QFile>`, `<QFileInfo>`
- Constructor: store cacheDir, call QDir().mkpath(m_cacheDir)
- Static fallback lists:
  ```cpp
  const QStringList LibraryAlbumArtProvider::kFrontFallbacks = {"cover.jpg", "Cover.jpg", "cover.png", "Cover.png", "folder.jpg", "Folder.jpg", "front.jpg", "Front.jpg", "front.png"};
  const QStringList LibraryAlbumArtProvider::kBackFallbacks = {"back.jpg", "Back.jpg", "back.png", "Back.png", "back-cover.jpg"};
  ```
- computeHash(): QCryptographicHash::hash((albumArtist + album).toUtf8(), QCryptographicHash::Sha1).toHex()
- hasCachedArt(): Check if front file exists in cache dir with any extension
- getCachedArt(): Return existing cached paths
- extractArt():
  1. Compute hash
  2. If hasCachedArt(), return getCachedArt() (skip re-extraction)
  3. Try extractFromFlac(flacPath, hash, "front")
  4. Try extractFromFlac(flacPath, hash, "back")
  5. If front empty: try findFileArt(albumDir, hash, "front")
  6. If back empty: try findFileArt(albumDir, hash, "back")
  7. Return {frontPath, backPath}
- extractFromFlac():
  - TagLib::FLAC::File flacFile(path.toUtf8().constData())
  - if !flacFile.isValid(), return empty
  - for pic in flacFile.pictureList():
    - if type=="front" && pic->type()==FrontCover, saveToDisk
    - if type=="back" && pic->type()==BackCover, saveToDisk
  - return path or empty
- findFileArt():
  - Get album directory from QFileInfo(flacPath).absolutePath()
  - For each filename in kFrontFallbacks (or kBackFallbacks):
    - If QFile::exists(albumDir + "/" + filename):
      - Read file, copy to cache dir with hash name
      - return cached path
  - return empty
- saveToDisk(): Same pattern as CdAlbumArtProvider -- detect extension from magic bytes, write to cacheDir/hash_type.ext
- detectExtension(): Check PNG magic (\x89PNG), return "png" or "jpg"
