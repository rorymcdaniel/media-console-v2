Create src/library/FlacLibraryController.h:
```cpp
#pragma once

#include <QObject>
#include <QVector>

#include <memory>

#include "library/LibraryDatabase.h"

struct LibraryConfig;
class LibraryScanner;
class LibraryAlbumArtProvider;
class LibraryArtistModel;
class LibraryAlbumModel;
class LibraryTrackModel;
class LocalPlaybackController;
class PlaybackState;

/// Lifecycle orchestrator for the FLAC library subsystem.
/// Coordinates scanning, database, art extraction, browse models, and playlist playback.
/// Follows CdController pattern: owns sub-components, wires signals, manages lifecycle.
class FlacLibraryController : public QObject
{
    Q_OBJECT
public:
    FlacLibraryController(LocalPlaybackController* playbackController,
                          PlaybackState* playbackState,
                          const LibraryConfig& config,
                          QObject* parent = nullptr);
    ~FlacLibraryController() override;

    void start();
    void stop();

    // User-initiated playback (FLAC-08)
    void playTrack(int trackModelIndex);
    void next();
    void previous();

    // Accessors for models (QML binding in Phase 10)
    LibraryArtistModel* artistModel() const;
    LibraryAlbumModel* albumModel() const;
    LibraryTrackModel* trackModel() const;

    bool isScanning() const;

signals:
    void scanProgress(int processed, int total);
    void scanComplete(int totalProcessed, int totalSkipped);
    void libraryReady();

private slots:
    void onBatchReady(const QVector<LibraryTrack>& tracks);
    void onScanComplete(int totalProcessed, int totalSkipped);
    void onTrackFinished();

private:
    void startScan();
    void playCurrentTrack();
    void updatePlaybackState();

    LocalPlaybackController* m_playbackController; // Non-owning
    PlaybackState* m_playbackState; // Non-owning
    QString m_rootPath;

    std::unique_ptr<LibraryDatabase> m_database;
    std::unique_ptr<LibraryScanner> m_scanner;
    std::unique_ptr<LibraryAlbumArtProvider> m_artProvider;
    std::unique_ptr<LibraryArtistModel> m_artistModel;
    std::unique_ptr<LibraryAlbumModel> m_albumModel;
    std::unique_ptr<LibraryTrackModel> m_trackModel;

    // Playlist state
    QVector<LibraryTrack> m_playlist;
    int m_currentIndex = -1;

    static constexpr int kPreviousRestartThresholdMs = 3000;
};
```

Create src/library/FlacLibraryController.cpp:
- Constructor:
  - Store non-owning pointers, rootPath from config
  - Create all owned components:
    ```cpp
    m_database = std::make_unique<LibraryDatabase>(this);
    m_scanner = std::make_unique<LibraryScanner>(this);
    QString cacheDir = QStandardPaths::writableLocation(QStandardPaths::CacheLocation) + "/album-art";
    m_artProvider = std::make_unique<LibraryAlbumArtProvider>(cacheDir, this);
    m_artistModel = std::make_unique<LibraryArtistModel>(m_database.get(), this);
    m_albumModel = std::make_unique<LibraryAlbumModel>(m_database.get(), m_artProvider.get(), this);
    m_trackModel = std::make_unique<LibraryTrackModel>(m_database.get(), this);
    ```
  - Wire signals:
    ```cpp
    connect(m_scanner.get(), &LibraryScanner::batchReady, this, &FlacLibraryController::onBatchReady);
    connect(m_scanner.get(), &LibraryScanner::scanComplete, this, &FlacLibraryController::onScanComplete);
    connect(m_scanner.get(), &LibraryScanner::scanProgress, this, &FlacLibraryController::scanProgress);
    if (m_playbackController)
        connect(m_playbackController, &LocalPlaybackController::trackFinished, this, &FlacLibraryController::onTrackFinished);
    ```
- start():
  - Open database: `m_database->open(QStandardPaths::writableLocation(QStandardPaths::CacheLocation) + "/library.db");`
  - Refresh models with existing data (library browsable immediately)
  - Start incremental scan
- startScan():
  - Build existingMtimes map by querying all file paths from database (or pass empty for first scan)
  - Note: For efficiency, add LibraryDatabase::getAllMtimes() method that returns QMap<QString, qint64>
  - `m_scanner->startScan(m_rootPath, existingMtimes);`
- onBatchReady(tracks):
  - Extract art for each unique album in batch (deduplicate by artist+album)
  - `m_database->upsertTrackBatch(tracks);`
  - Refresh artist model (albums and tracks refresh on filter change)
  - `m_artistModel->refresh();`
- onScanComplete(totalProcessed, totalSkipped):
  - Get set of valid file paths from database, remove stale entries
  - Final model refresh
  - emit libraryReady()
  - emit scanComplete(totalProcessed, totalSkipped)
  - qCInfo(mediaLibrary) << "Library scan complete:" << totalProcessed << "processed," << totalSkipped << "skipped"
- playTrack(trackModelIndex):
  - Load album track list from track model: `m_playlist = m_trackModel->allTracks();`
  - Set m_currentIndex = trackModelIndex (bounds check)
  - playCurrentTrack()
- playCurrentTrack():
  - Get track: `const auto& track = m_playlist[m_currentIndex];`
  - Create stream: `auto stream = std::make_shared<FlacAudioStream>(track.filePath);`
  - Update PlaybackState:
    ```cpp
    m_playbackState->setActiveSource(MediaSource::Library);
    m_playbackState->setTitle(track.title);
    m_playbackState->setArtist(track.artist);
    m_playbackState->setAlbum(track.album);
    m_playbackState->setTrackNumber(m_currentIndex + 1);
    m_playbackState->setTrackCount(m_playlist.size());
    m_playbackState->setDurationMs(track.durationSeconds * 1000LL);
    ```
  - Set album art: `auto art = m_artProvider->getCachedArt(track.albumArtist, track.album); m_playbackState->setAlbumArtUrl(art.frontPath.isEmpty() ? "" : "file://" + art.frontPath);`
  - Play: `m_playbackController->play(stream);`
- next():
  - If m_currentIndex + 1 < m_playlist.size(): increment, playCurrentTrack()
  - Else: stop playback (album finished), clear playlist
- previous():
  - If current position > kPreviousRestartThresholdMs: seek to 0 (restart current)
  - Else if m_currentIndex > 0: decrement, playCurrentTrack()
  - Else: seek to 0 (restart first track)
- onTrackFinished():
  - If active source is Library (check m_playbackState->activeSource()): call next()
  - This way CD subsystem's trackFinished doesn't trigger library next
- stop():
  - m_scanner->cancel()
  - Clear playlist
  - m_currentIndex = -1
- Destructor: call stop()
