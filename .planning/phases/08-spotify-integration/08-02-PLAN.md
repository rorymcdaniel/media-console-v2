Output: SpotifyClient class with methods for search, devices, playback control, queue, playlists, and current playback state.

Spotify Web API endpoints (from research):
```
Search:          GET  /v1/search?q={q}&type={type}&limit={limit}
Get Devices:     GET  /v1/me/player/devices
Get Playback:    GET  /v1/me/player
Transfer:        PUT  /v1/me/player                    body: {device_ids, play}
Play:            PUT  /v1/me/player/play               body: {uris} or {context_uri}
Pause:           PUT  /v1/me/player/pause
Next:            POST /v1/me/player/next
Previous:        POST /v1/me/player/previous
Add to Queue:    POST /v1/me/player/queue?uri={uri}
User Playlists:  GET  /v1/me/playlists?limit={limit}
Featured:        GET  /v1/browse/featured-playlists?limit={limit}
Playlist Tracks: GET  /v1/playlists/{id}/tracks
Artist Top:      GET  /v1/artists/{id}/top-tracks?market=US
Album Tracks:    GET  /v1/albums/{id}/tracks
```

Base URL: https://api.spotify.com
All endpoints need header: Authorization: Bearer {access_token}

    1. Create src/spotify/SpotifyClient.h:
       ```cpp
       #pragma once

       #include <QJsonArray>
       #include <QJsonObject>
       #include <QNetworkAccessManager>
       #include <QObject>

       class SpotifyClient : public QObject
       {
           Q_OBJECT

       public:
           explicit SpotifyClient(QObject* parent = nullptr);

           /// Set the access token for API calls. Called by SpotifyController
           /// whenever SpotifyAuth provides a new token.
           void setAccessToken(const QString& token);

           // --- Search ---
           void search(const QString& query, int limit = 10);

           // --- Devices ---
           void getDevices();

           // --- Playback state ---
           void getCurrentPlayback();

           // --- Playback control ---
           void transferPlayback(const QString& deviceId, bool startPlaying = true);
           void play(const QString& deviceId = QString());
           void playUri(const QString& uri, const QString& deviceId = QString());
           void playContext(const QString& contextUri, int offsetIndex = 0, const QString& deviceId = QString());
           void pause(const QString& deviceId = QString());
           void next(const QString& deviceId = QString());
           void previous(const QString& deviceId = QString());

           // --- Queue ---
           void addToQueue(const QString& trackUri, const QString& deviceId = QString());

           // --- Browse ---
           void getUserPlaylists(int limit = 20);
           void getFeaturedPlaylists(int limit = 20);
           void getPlaylistTracks(const QString& playlistId);
           void getArtistTopTracks(const QString& artistId, const QString& market = "US");
           void getAlbumTracks(const QString& albumId);

           /// Build a full API URL (exposed for testing).
           static QUrl buildUrl(const QString& endpoint, const QList<QPair<QString, QString>>& params = {});

       signals:
           // Search results
           void searchResultsReady(const QJsonObject& results);
           void searchFailed(const QString& error, int statusCode);

           // Device list
           void devicesReady(const QJsonArray& devices);
           void devicesFailed(const QString& error, int statusCode);

           // Current playback
           void currentPlaybackReady(const QJsonObject& playback);
           void noActivePlayback(); // 204 response — nothing playing
           void currentPlaybackFailed(const QString& error, int statusCode);

           // Playback control responses
           void playbackCommandSucceeded(const QString& command);
           void playbackCommandFailed(const QString& command, const QString& error, int statusCode);

           // Transfer
           void transferSucceeded();
           void transferFailed(const QString& error, int statusCode);

           // Queue
           void addToQueueSucceeded();
           void addToQueueFailed(const QString& error, int statusCode);

           // Browse/playlists
           void userPlaylistsReady(const QJsonObject& playlists);
           void featuredPlaylistsReady(const QJsonObject& playlists);
           void playlistTracksReady(const QJsonArray& tracks);
           void artistTopTracksReady(const QJsonArray& tracks);
           void albumTracksReady(const QJsonArray& tracks);
           void browseFailed(const QString& error, int statusCode);

       private:
           QNetworkRequest createRequest(const QUrl& url) const;

           /// Handle a reply that returns JSON object (search, playback, playlists).
           void handleJsonObjectReply(QNetworkReply* reply,
                                      const std::function<void(const QJsonObject&)>& onSuccess,
                                      const std::function<void(const QString&, int)>& onError);

           /// Handle a reply that returns no content (204) on success (play, pause, etc).
           void handleCommandReply(QNetworkReply* reply, const QString& commandName);

           QNetworkAccessManager m_networkManager;
           QString m_accessToken;

           static constexpr const char* kBaseUrl = "https://api.spotify.com/v1/";
       };
       ```

    2. Create src/spotify/SpotifyClient.cpp:
       - buildUrl(): Construct QUrl from kBaseUrl + endpoint, apply QUrlQuery params
       - createRequest(): Set Authorization header ("Bearer " + m_accessToken), Content-Type: application/json
       - setAccessToken(): Store token

       Search implementation:
       - GET /v1/search with q, type=track,artist,album, limit params
       - Parse response JSON, emit searchResultsReady with the full JSON object
       - On error: emit searchFailed with error string and HTTP status code

       Device implementation:
       - GET /v1/me/player/devices
       - Parse "devices" array from response, emit devicesReady
       - On error: emit devicesFailed

       Current playback:
       - GET /v1/me/player
       - Handle 204 (no content) -> emit noActivePlayback()
       - Handle 200 -> emit currentPlaybackReady with JSON
       - On error: emit currentPlaybackFailed

       Transfer playback:
       - PUT /v1/me/player with body {"device_ids": [deviceId], "play": startPlaying}
       - On 204 success: emit transferSucceeded
       - On error: emit transferFailed

       Play/pause/next/previous:
       - play(): PUT /v1/me/player/play with optional device_id query param
       - playUri(): PUT /v1/me/player/play with body {"uris": [uri]} and optional device_id
       - playContext(): PUT /v1/me/player/play with body {"context_uri": contextUri, "offset": {"position": offsetIndex}} and optional device_id
       - pause(): PUT /v1/me/player/pause with optional device_id
       - next(): POST /v1/me/player/next with optional device_id
       - previous(): POST /v1/me/player/previous with optional device_id
       - All use handleCommandReply for 204 success / error handling

       Queue:
       - POST /v1/me/player/queue with uri query param and optional device_id
       - On 204: emit addToQueueSucceeded
       - On error: emit addToQueueFailed

       Browse/playlists:
       - getUserPlaylists(): GET /v1/me/playlists with limit
       - getFeaturedPlaylists(): GET /v1/browse/featured-playlists with limit
       - getPlaylistTracks(): GET /v1/playlists/{id}/tracks
       - getArtistTopTracks(): GET /v1/artists/{id}/top-tracks with market
       - getAlbumTracks(): GET /v1/albums/{id}/tracks
       - All parse appropriate JSON and emit signals

       handleJsonObjectReply helper:
       - Connect to QNetworkReply::finished
       - Check error, parse JSON, call onSuccess or onError
       - Always deleteLater on reply

       handleCommandReply helper:
       - Connect to QNetworkReply::finished
       - 204 = success, emit playbackCommandSucceeded(commandName)
       - Other = error, emit playbackCommandFailed(commandName, error, status)

       Use mediaSpotify logging category. Log all API calls at debug level, errors at warning level.
