AppBuilder wiring makes Spotify available to the full application.

Output: SpotifyController class, AppBuilder integration, AppContext with SpotifyController pointer.

       ```cpp
       #pragma once

       #include <QJsonArray>
       #include <QJsonObject>
       #include <QObject>
       #include <QTimer>

       class SpotifyAuth;
       class SpotifyClient;
       class PlaybackState;
       class UIState;

       struct SpotifyConfig;

       class SpotifyController : public QObject
       {
           Q_OBJECT

           // Properties for QML binding (Phase 10)
           Q_PROPERTY(bool spotifyAvailable READ isSpotifyAvailable NOTIFY spotifyAvailableChanged)
           Q_PROPERTY(bool spotifyActive READ isSpotifyActive NOTIFY spotifyActiveChanged)
           Q_PROPERTY(QJsonObject searchResults READ searchResults NOTIFY searchResultsChanged)
           Q_PROPERTY(QJsonArray suggestedPlaylists READ suggestedPlaylists NOTIFY suggestedPlaylistsChanged)

       public:
           SpotifyController(SpotifyAuth* auth, SpotifyClient* client,
                             PlaybackState* playbackState, UIState* uiState,
                             const SpotifyConfig& config, QObject* parent = nullptr);

           /// True if Spotify is authenticated and tokens are valid.
           bool isSpotifyAvailable() const;

           /// True if Spotify is the active playback source.
           bool isSpotifyActive() const;

           /// Current search results (for QML binding).
           QJsonObject searchResults() const;

           /// Suggested playlists for default content (for QML binding).
           QJsonArray suggestedPlaylists() const;

       public slots:
           // --- Source activation (called when user selects Spotify) ---

           /// Called when user selects Spotify as source. Checks auth state,
           /// detects active sessions, handles transfer.
           void activateSpotify();

           /// Called when user switches away from Spotify source.
           void deactivateSpotify();

           // --- Search ---

           /// Debounced search. Resets timer on each call, fires after 300ms.
           void search(const QString& query);

           /// Clear search results (when user clears search field).
           void clearSearch();

           // --- Playback control ---
           void play();
           void pause();
           void next();
           void previous();
           void playTrackUri(const QString& uri);
           void playAlbumUri(const QString& albumUri, int trackIndex = 0);
           void addToQueue(const QString& trackUri);

           // --- Session takeover ---

           /// User confirmed transfer from takeover dialog.
           void confirmTransfer();

           /// User canceled transfer from takeover dialog.
           void cancelTransfer();

           // --- Browse drill-down ---
           void loadArtistTopTracks(const QString& artistId);
           void loadAlbumTracks(const QString& albumId);
           void loadPlaylistTracks(const QString& playlistId);

       signals:
           void spotifyAvailableChanged(bool available);
           void spotifyActiveChanged(bool active);
           void searchResultsChanged(const QJsonObject& results);
           void suggestedPlaylistsChanged(const QJsonArray& playlists);

           /// Emitted when an active session is detected on another device.
           /// QML shows the takeover dialog with this info.
           void activeSessionDetected(const QString& deviceName,
                                      const QString& trackTitle,
                                      const QString& artistName);

           /// Emitted when artist/album/playlist tracks are loaded for drill-down.
           void drillDownTracksReady(const QJsonArray& tracks, const QString& title);

           /// Emitted on errors that should be shown to the user.
           void spotifyError(const QString& message);

       private slots:
           void onSearchDebounceTimeout();
           void onSearchResultsReady(const QJsonObject& results);
           void onDevicesReady(const QJsonArray& devices);
           void onCurrentPlaybackReady(const QJsonObject& playback);
           void onNoActivePlayback();
           void onTransferSucceeded();
           void onTransferFailed(const QString& error, int statusCode);
           void onPlaybackPollTimeout();
           void onPlaybackCommandFailed(const QString& command, const QString& error, int statusCode);
           void onAuthStateChanged(bool authenticated);
           void onUserPlaylistsReady(const QJsonObject& playlists);
           void onFeaturedPlaylistsReady(const QJsonObject& playlists);

       private:
           void startPlaybackPolling();
           void stopPlaybackPolling();
           void updatePlaybackStateFromJson(const QJsonObject& playback);
           void findAndTransferToDevice();
           void loadSuggestedContent();

           SpotifyAuth* m_auth;
           SpotifyClient* m_client;
           PlaybackState* m_playbackState;
           UIState* m_uiState;
           QString m_desiredDeviceName;

           // Search debounce
           QTimer m_searchDebounceTimer;
           QString m_pendingSearchQuery;

           // Playback state polling
           QTimer m_playbackPollTimer;
           bool m_spotifyActive = false;

           // Session takeover state
           QString m_pendingTransferDeviceId;

           // Cached data
           QJsonObject m_searchResults;
           QJsonArray m_suggestedPlaylists;

           static constexpr int kSearchDebounceMs = 300;
           static constexpr int kPlaybackPollMs = 3000;
       };
       ```

    2. Create src/spotify/SpotifyController.cpp:
       Constructor:
       - Store pointers, m_desiredDeviceName from config.desiredDeviceName
       - Configure m_searchDebounceTimer: singleShot, kSearchDebounceMs, connect to onSearchDebounceTimeout
       - Connect all SpotifyClient signals to controller slots
       - Connect SpotifyAuth::authStateChanged to onAuthStateChanged
       - Configure m_playbackPollTimer: kPlaybackPollMs interval, connect to onPlaybackPollTimeout
       - Call m_client->setAccessToken(m_auth->accessToken()) if authenticated
       - Connect auth granted signal to update client token

       activateSpotify():
       - If not authenticated: emit UIState::showToast("Spotify not configured — run media-console --spotify-auth", "error"), return
       - Set m_spotifyActive = true, emit spotifyActiveChanged
       - Update client access token from auth
       - Call m_client->getCurrentPlayback() to check for active session
       - Load suggested content

       deactivateSpotify():
       - m_spotifyActive = false, emit spotifyActiveChanged
       - stopPlaybackPolling()

       search():
       - Store query in m_pendingSearchQuery
       - Restart m_searchDebounceTimer (stops any pending, starts fresh 300ms)

       clearSearch():
       - m_searchDebounceTimer.stop()
       - m_searchResults = QJsonObject()
       - emit searchResultsChanged

       onSearchDebounceTimeout():
       - If m_pendingSearchQuery is empty, return
       - Call m_client->search(m_pendingSearchQuery, 10)

       onSearchResultsReady():
       - Store in m_searchResults, emit searchResultsChanged

       Play/pause/next/previous:
       - Delegate to m_client with empty deviceId (uses active device)
       - Log via mediaSpotify

       playTrackUri(): m_client->playUri(uri)
       playAlbumUri(): m_client->playContext(albumUri, trackIndex)
       addToQueue(): m_client->addToQueue(trackUri)

       onCurrentPlaybackReady():
       - Parse JSON for device name, is_playing, item (track info)
       - If playback is active on a DIFFERENT device (device.name != m_desiredDeviceName):
         Emit activeSessionDetected(deviceName, trackTitle, artistName)
         Store device info for potential transfer
       - If playback is on our device or no playback:
         Start polling, update PlaybackState

       onNoActivePlayback():
       - No session to take over
       - findAndTransferToDevice() to start fresh

       confirmTransfer():
       - Call m_client->transferPlayback(m_pendingTransferDeviceId, true)

       cancelTransfer():
       - Clear m_pendingTransferDeviceId
       - User stays on current view

       findAndTransferToDevice():
       - Call m_client->getDevices()
       - onDevicesReady: find device matching m_desiredDeviceName
       - If found: call m_client->transferPlayback(deviceId, false)
       - If not found: emit spotifyError("Device 'Voice of Music' not found. Ensure the receiver is on and connected.")

       onTransferSucceeded():
       - startPlaybackPolling()
       - Log success

       onTransferFailed():
       - emit spotifyError with user-friendly message
       - If 403: "Spotify Premium required for playback control"

       startPlaybackPolling():
       - Start m_playbackPollTimer if not running

       stopPlaybackPolling():
       - Stop m_playbackPollTimer

       onPlaybackPollTimeout():
       - If not m_spotifyActive, stop and return
       - Call m_client->getCurrentPlayback()

       updatePlaybackStateFromJson():
       - Extract from JSON: item.name (title), item.artists[0].name (artist), item.album.name (album), item.duration_ms, progress_ms, is_playing
       - For album art (SPOT-08): Do NOT use Spotify API image URL for active playback. The receiver provides album art via its CGI endpoint (already parsed as NJA2 by ReceiverController). Only use Spotify images for search results.
       - Write to m_playbackState: setTitle, setArtist, setAlbum, setPositionMs, setDurationMs, setPlaybackMode(is_playing ? Playing : Paused)
       - Do NOT set albumArtUrl here — receiver handles it via ReceiverController NJA2 parsing

       onPlaybackCommandFailed():
       - If 403: showToast "Spotify Premium required"
       - If 429: showToast "Too many requests, try again"
       - Otherwise: showToast with error message

       onAuthStateChanged():
       - emit spotifyAvailableChanged
       - If authenticated: update client token
       - If not authenticated and spotifyActive: deactivateSpotify, showToast

       loadSuggestedContent():
       - Try m_client->getFeaturedPlaylists() first
       - On success: parse playlists items into m_suggestedPlaylists
       - On browse fail (403): fall back to m_client->getUserPlaylists()
       - onUserPlaylistsReady: parse into m_suggestedPlaylists

       loadArtistTopTracks/loadAlbumTracks/loadPlaylistTracks:
       - Delegate to m_client methods
       - Connect to drillDownTracksReady signal

       Use mediaSpotify logging category throughout.
