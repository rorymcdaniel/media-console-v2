    ```qml
        var m = Math.floor(seconds / 60)
        var s = seconds % 60
        return m + ":" + (s < 10 ? "0" : "") + s
    }
    ```

    Add the CD track list after the PlaybackControls Rectangle (after line ~334):
    ```qml
    // ========== CD Track List (UI-09) ==========
    // Visible when CD is active source and NOT currently playing
    Column {
        id: cdTrackList
        width: parent.width
        spacing: 0
        visible: PlaybackState.activeSource === MediaSource.CD
                 && PlaybackState.playbackMode !== PlaybackMode.Playing

        Text {
            text: "Select a Track"
            color: Theme.textSecondary
            font.pixelSize: Theme.fontSizeSmall
            font.bold: true
            bottomPadding: Theme.spacingSmall
        }

        Repeater {
            model: CdController.toc

            delegate: Rectangle {
                width: cdTrackList.width
                height: Theme.touchTargetSmall
                color: cdTrackItemMa.pressed ? Theme.glassBg : "transparent"
                radius: Theme.radiusSmall

                Row {
                    anchors.verticalCenter: parent.verticalCenter
                    anchors.left: parent.left
                    anchors.right: parent.right
                    anchors.leftMargin: Theme.spacingSmall
                    anchors.rightMargin: Theme.spacingSmall
                    spacing: Theme.spacingSmall

                    Text {
                        text: modelData.trackNumber + "."
                        color: Theme.textDimmed
                        font.pixelSize: Theme.fontSizeSmall
                        width: 24
                    }

                    Text {
                        text: modelData.title !== "" ? modelData.title
                                                    : ("Track " + modelData.trackNumber)
                        color: Theme.textPrimary
                        font.pixelSize: Theme.fontSizeSmall
                        elide: Text.ElideRight
                        width: parent.width - 80
                    }

                    Text {
                        text: root.formatDuration(modelData.durationSeconds)
                        color: Theme.textDimmed
                        font.pixelSize: Theme.fontSizeSmall
                        anchors.right: parent.right
                    }
                }

                MouseArea {
                    id: cdTrackItemMa
                    anchors.fill: parent
                    onClicked: {
                        CdController.playTrack(modelData.trackNumber)
                        UIState.setActiveView(ActiveView.NowPlaying)
                    }
                }
            }
        }
    }
    ```

    IMPORTANT implementation notes:
    - `CdController.toc` is the QVariantList Q_PROPERTY from Plan 02 — each element is a QVariantMap

    - Access map fields via `modelData.trackNumber`, `modelData.durationSeconds`, `modelData.title`
    - `PlaybackMode.Playing` is the enum value — the list hides as soon as playback starts
    - `UIState.setActiveView(ActiveView.NowPlaying)` already used in LibraryBrowser.qml:526 — same pattern
    - The track list is inside the right Column which has `width: parent.width - Theme.spacingLarge * 2`
    - Do not add a ScrollView or ListView — Repeater is sufficient for max ~20 tracks
    - The track list appearing/disappearing is handled by the visible property — no animation needed
